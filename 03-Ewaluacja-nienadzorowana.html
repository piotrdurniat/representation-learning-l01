def compute_cov_eig(z: Tensor) -> Tensor:
    covariance_matrix = torch.cov(z.t())
    eigenvalues, _ = torch.linalg.eig(covariance_matrix)
    return eigenvalues

Zadanie 6.3 (0.5 pkt)

Wyej otrzymane spektrum moe posuy analizie i porwnaniom rnych reprezentacji. Czsto jednak zamiast spektrum chcielibymy otrzyma pojedyncz warto jako miar jakoci reprezentacji. Przykadowo powysz krzyw mona dopasowa do rokadu power-law ($\lambda_j \propto j^{-\alpha},$ gdzie $\lambda_j$ jest $j$-t wartoci wasn, a $\alpha$ wspczynnikiem zanikania (ang. coefficient of decay)), a otrzymany wspczynnik potraktowa jako tak miar (Gosh et al., 2022).

Duo prostszym i jednoczenie skuteczniejszym podejciem jest metoda $\texttt{RankMe}$ (Garrido et al., 2022), ktra pozwala na estymacje rzdu (ang. rank) macierzy reprezetacji na podstawie entropii rozkadu wartoci osobliwych (ang. signular values z metody Singular Value Deocomposition (SVD)), co sprowadza si do poniszego wzoru: $$\texttt{RankMe}(\boldsymbol{Z}) = exp \left(- \sum_{k=1}^{min(N, D)} p_klog p_k \right)$$ $$p_k = \frac{\sigma_k(\boldsymbol{Z})}{\lVert \sigma_k(\boldsymbol{Z}) \rVert_1} + \epsilon$$ gdzie $\boldsymbol{Z}$ - macierz reprezentacji $(N, D)$, $\sigma_k(\boldsymbol{Z}$ - $k$-ta warto osobliwa, $\epsilon$ - maa warto dla stabilnoci oblicze

Zaimplementuj funkcj rank_me wykorzystujc powyszy wzr. Wymagane jest pene zrozumienie metody, w szczeglnoci poj: SVD, entropii oraz rzdu macierzy.

def rank_me(Z, epsilon=1e-10) -> float:
    singular_values = np.linalg.svd(Z, compute_uv=False)
    p_k = singular_values / np.sum(singular_values) + epsilon

    entropy = -np.sum(p_k * np.log(p_k))
    rank_me_value = np.exp(entropy)

    return rank_me_value

from torch import nn


class Autoencoder(nn.Module):

    def __init__(self, feature_dim: int, hidden_dim: int):
        super().__init__()
        self.encoder = nn.Sequential(
            nn.Linear(feature_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
        )
        self.decoder = nn.Sequential(
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, feature_dim),
        )

    def forward(self, x):
        return self.decoder(self.encoder(x)), self.encoder(x)

import torch
from torch.nn import functional as F


def train(
    model: Autoencoder,
    optimizer: torch.optim.Optimizer,
    x: torch.Tensor,
) -> tuple[float, float]:
    model.train()

    optimizer.zero_grad()

    x_rec, z = model(x)
    loss = F.mse_loss(input=x_rec, target=x)

    loss.backward()
    optimizer.step()

    rank = rank_me(z.detach().numpy())

    return loss.item(), rank.item()


def test(model: Autoencoder, x: torch.Tensor) -> float:
    model.eval()

    with torch.no_grad():
        x_rec, _ = model(x)
        loss = F.mse_loss(input=x_rec, target=x)

        return loss.item()
